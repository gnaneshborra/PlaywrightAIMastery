26 - Arithmetic Operators

26_Assigned_Operator.js Arithmetic

Covers all arithmetic operators: addition (+), subtraction (-), multiplication (*), division (/), modulus (% for remainder), and exponentiation (**). The modulus operator is especially useful for checking odd/even numbers (n % 2).

const { log } = require("node:console");

let a = 10, b = 3;
console.log(a);
console.log(b);

// Arithmetic Operators
let sum = a + b;
let sub = a - b;
let mul = a * b;
let div = a / b;

console.log(sum);
console.log(sub);
console.log(mul);
console.log(div);

// -------
console.log(a % b); // Modulus operator - it gives -> Remainder
console.log(13 % 7);

console.log(100 % 2);
console.log(101 % 2);
console.log(103 % 2);

//  n%2 == 1 - odd number, [ n%2 == 0 , even]

console.log(2 ** 3); // exponential a^b
console.log(a ** b); // exponential a^b 10^3

// Increment and Decrement Operators
// a++, b-- (learn about them, one more class)
let firstname = "Pramod";

27 - Assignment Operators

27_Assignment_Operators.js Assignment

Assignment operators provide shorthand for updating variables: += (add and assign), -= (subtract and assign), *= (multiply and assign), /= (divide and assign), and %= (modulus and assign).

let x = 10;
//x = x + 10;
x += 10; // --> //x = x + 10;
x -= 3; // x = x - 3;
console.log(x);
x *= 2; // x = x * 2;
console.log(x);
x /= 4;  // x = x/4;
console.log(x);
x %= 4;
console.log(x);

28 - Comparison Operators

28_Comparsion_Operators.js Comparison

Comparison operators always return a boolean. Covers >, <, >=, <=, == (loose), === (strict), != (loose not equal), and !== (strict not equal). Strict checks both data type and value.

// Comparison Operator (will always result in boolean, true or false)

// = -> Assignment operator
// == -> loose comparison
// === -> strict comparison

// > , < , >= , <= , !, !==

console.log(3 > 4);
console.log(3 < 4);
console.log(4 >= 4); // 4 > 4 or 4===4 , or gate
console.log(3 <= 4);

console.log(5 == "5"); // loose comparison
console.log(5 === "5"); // strict not allowed

console.log(5 != "5"); // false - loose
console.log(5 !== "5"); // true

// === Strict check: both datatype and value
// == Loose check: either value or data type

29 - Logical Operators

29_Logical_Operators.js Logical

The three logical operators: && (AND -- both must be true), || (OR -- at least one must be true), and ! (NOT -- reverses the boolean value).

// && -> AND Gate
// || -> OR Gate

let a = true;
let b = false;
console.log(a && b); //  AND
console.log(a || b); //  OR
console.log(!a); // Reverse of a

30 - String Operators

30_String_Operators.js Strings

The += operator can be used with strings for concatenation. It appends the right-side string to the existing string variable.

let s = "Hi";
s += " Dev";
console.log(s);

31 - Ternary Operator

31_Ternary_Operators.js Ternary

The ternary operator is a shorthand for if/else: condition ? trueResult : falseResult. Useful for simple conditional assignments in a single line.

let age = 20;

let is_pramod_will_go_to_goa = age > 18 ? "Yes, let go goa!" : "No you are minor, Not going";
console.log(is_pramod_will_go_to_goa)
// Ternary = condition ? "true result" : "false result";

31 - Type Operators (typeof)

31_Type_Operators.js typeof

The typeof operator returns a string indicating the type of a value. Note the quirks: typeof null returns "object" (a known JS bug), and typeof [] also returns "object".

console.log(typeof "hello");
console.log(typeof 123); // int -> number
console.log(typeof 31.4); // float -> number
// typeof true
// typeof undefined  -> undefined
// typeof null  -> object
// typeof [] -> object
console.log(typeof []); // -> object

console.log(`${6 * 4}`);

32 - Nullish Coalescing Operator (??)

32_Null_Optinal_Value.js Modern JS

The nullish coalescing operator (??) returns the right-hand operand when the left-hand operand is null or undefined. Unlike ||, it does NOT treat 0, "", or false as fallback triggers. Very useful in test data handling and modern TypeScript.

let amul = null;
let val = amul ?? "nandani milk";
let val2 = null ?? "default";   // "default" (?? returns right side if left is null/undefined)

// Modern JS - Typescript.

val = "which milk? -> " + val;
console.log(val)
// very useful in test data handling.
